🖥️ Smoke Test — pegar en consola del sitio real
(function tmSmoke(){
  const ok = (m)=>console.log('%c✓ '+m,'color:#27ae60;font-weight:700');
  const fail = (m,e)=>console.error('%c✗ '+m+': '+e,'color:#e74c3c;font-weight:700');

  // DOM
  ['categoriasGrid','productosGrid','heroSearchInput','productDetailModal',
   'carritoDrawer','notifModal'].forEach(id=>{
    document.getElementById(id) ? ok('DOM: #'+id) : fail('DOM: #'+id,'no encontrado');
  });

  // Funciones globales
  ['agregarAlCarrito','mostrarVistaCategoria','abrirDetalleProducto',
   'buscarDesdeHero','renderizarProductos','escapeHtml'].forEach(fn=>{
    typeof window[fn]==='function' ? ok('fn: '+fn+'()') : fail('fn: '+fn+'()','no definida');
  });

  // SW
  navigator.serviceWorker.getRegistration('/').then(r=>{
    r ? ok('SW registrado: '+r.scope) : fail('SW','no registrado');
  });

  // productos
  typeof productos!=='undefined'&&Array.isArray(productos)&&productos.length>0
    ? ok('productos cargados: '+productos.length+' items')
    : fail('productos','no cargados o vacíos');

  // localStorage
  try{ localStorage.setItem('_smoke','1'); localStorage.removeItem('_smoke'); ok('localStorage OK'); }
  catch(e){ fail('localStorage',e.message); }

  // Firebase
  typeof firebase!=='undefined' ? ok('Firebase SDK cargado') : fail('Firebase','SDK no encontrado');

  console.log('%c\n▶ Smoke test completo','color:#c9a96e;font-weight:800');
})();
